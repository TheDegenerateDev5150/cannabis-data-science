\begin{frame}{A couple concepts from economics}

{\bfseries 1. Expected Utility} -- A utility function that is linear in probabilities

\vspace{-0.5\baselineskip}
$$
U(L) = \sum_{s=1}^{S}p_sU(c_s)
$$

\vspace{1\baselineskip}

{\bfseries 2. Cobb--Douglas Utility Function} -- A utility function where

\vspace{-0.5\baselineskip}
$$
U(x_{1},\dots,x_{L}) = \alpha_1\log x_{1} + \cdots + \alpha_{L}\log x_{L},
$$

\vspace{0.75\baselineskip}

for some $\alpha_{l} > 0$ such that $\sum_{l=1}^L \alpha_{l} = 1$.

\end{frame}
