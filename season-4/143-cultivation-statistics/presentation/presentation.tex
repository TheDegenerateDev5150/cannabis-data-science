% Conclusion.
\begin{frame}{The Benefit of Using Banned Pesticides}

Given
\vspace{0.5\baselineskip}

\begin{itemize}

\item $LTV_{plant}$ is the lifetime value of a plant.

\vspace{0.5\baselineskip}

\item $N_{plant}$ is the number of plants.

\vspace{0.5\baselineskip}

\item $\Delta Pr(\text{survival})_{pesticides}$ is the change in the survival rate of plants when pesticides are applied.

\vspace{0.5\baselineskip}

\end{itemize}

\vspace{1\baselineskip}

The {\bfseries benefit} to using banned pesticides is

$$
LTV_{plant} \times N_{plant} \times \Delta Pr(\text{survival})_{pesticides}
$$

\end{frame}


\begin{frame}{The Cost of Using Banned Pesticides}

Given
\vspace{0.5\baselineskip}

\begin{itemize}

\item $\text{Cost}_{pesticides}$ is the cost of banned pesticides.

\vspace{0.5\baselineskip}

\item $\text{Cost}_{\text{enforcement}}$ is the cost when enforcement happens

\vspace{0.5\baselineskip}

\item $Pr_{\text{enforcement}}$ is the probability of enforcement

\vspace{0.5\baselineskip}

\end{itemize}

\vspace{1\baselineskip}

The {\bfseries cost} to using banned pesticides is

$$
\text{Cost}_{pesticides} + Pr_{\text{enforcement}} \times \text{Cost}_{\text{enforcement}}
$$

\end{frame}


% Conclusion.
\begin{frame}{Why does this matter? The incentive to use pesticides}

Cultivators will {\bfseries not use} banned pesticides when*:

\vspace{-0.5\baselineskip}
$$
N_{plant} \times \Delta Pr(\text{s})_{pesticides} \leq Pr_{\text{enforcement}} \times \text{Cost}_{\text{enforcement}}
$$

\vspace{0.5\baselineskip}

Likelihood of use \underline{increases} when:

\begin{itemize}

    \vspace{0.5\baselineskip}

    \item $N_{plant}$ $\uparrow$
    
    \vspace{0.5\baselineskip}
    
    \item $\Delta Pr(\text{s})_{pesticides}$ $\uparrow$ 

\end{itemize} 
   
   
\vspace{1\baselineskip}
Likelihood of use \underline{decreases} when:

\begin{itemize}

    \vspace{0.5\baselineskip}
    
    \item $Pr_{\text{enforcement}}$ $\uparrow$
  
    \vspace{0.5\baselineskip}
    
    \item $\text{Cost}_{\text{enforcement}}$ $\uparrow$
\end{itemize}


\vfill
{\tiny * Assuming $\text{Cost}_{pesticides}$ is approximately 0 and $\text{Cost}_{enforcement}$ is normalized to $LTV_{plant}$.}


\end{frame}
